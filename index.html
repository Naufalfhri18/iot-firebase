Salin kode Arduino berikut untuk diunggah ke mikrokontroler ESP32 Anda menggunakan Arduino IDE:

#include <WiFi.h>
#include <FirebaseESP32.h>
#include "DHT.h"

#define WIFI_SSID "SSID_WIFI_MU"
#define WIFI_PASSWORD "PASSWORD_WIFI_MU"
#define FIREBASE_HOST "https://iot-firebase-6c9d2-default-rtdb.asia-southeast1.firebasedatabase.app"
#define FIREBASE_AUTH "AIzaSyD9yQhJYAb5hcfgaG9InroxUEVL7DImrLY"

#define RELAY1_PIN 26
#define RELAY2_PIN 25
#define RELAY3_PIN 27
#define RELAY4_PIN 14
#define DHTPIN 4
#define DHTTYPE DHT11

DHT dht(DHTPIN, DHTTYPE);
FirebaseData firebaseData;

void setup() {
  Serial.begin(115200);
  pinMode(RELAY1_PIN, OUTPUT);
  pinMode(RELAY2_PIN, OUTPUT);
  pinMode(RELAY3_PIN, OUTPUT);
  pinMode(RELAY4_PIN, OUTPUT);
  
  WiFi.begin(WIFI_SSID, WIFI_PASSWORD);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500); Serial.print(".");
  }
  Serial.println("WiFi Terkoneksi!");
  
  Firebase.begin(FIREBASE_HOST, FIREBASE_AUTH);
  dht.begin();
}

void loop() {
  // Transmit DHT11 Sensor data to Firebase Database
  float t = dht.readTemperature();
  float h = dht.readHumidity();
  if (!isnan(t) && !isnan(h)) {
    Firebase.setFloat(firebaseData, "/sensor/temperature", t);
    Firebase.setInt(firebaseData, "/sensor/humidity", h);
  }
  
  // Listen relay controller changes from Firebase Web client
  if (Firebase.getInt(firebaseData, "/relay/relay1")) {
    digitalWrite(RELAY1_PIN, firebaseData.intData());
  }
  if (Firebase.getInt(firebaseData, "/relay/relay2")) {
    digitalWrite(RELAY2_PIN, firebaseData.intData());
  }
  if (Firebase.getInt(firebaseData, "/relay/relay3")) {
    digitalWrite(RELAY3_PIN, firebaseData.intData());
  }
  if (Firebase.getInt(firebaseData, "/relay/relay4")) {
    digitalWrite(RELAY4_PIN, firebaseData.intData());
  }
  delay(2000);
}

Struktur model JSON yang dipakai di Firebase Realtime Database Anda adalah sebagai berikut:

{
  "relay": {
    "relay1": 0,
    "relay2": 0,
    "relay3": 0,
    "relay4": 0
  },
  "sensor": {
    "temperature": 27.5,
    "humidity": 64
  }
}

Pastikan juga Rules (Aturan Keamanan) di menu Realtime Database Firebase Anda telah disetel publik saat masa pengembangan kampus:

{
  "rules": {
    ".read": true,
    ".write": true
  }
}

Untuk mendeploy project website Anda ke Firebase Hosting agar dapat diakses dari luar jaringan secara online:

  • 1. Install Firebase CLI di laptop:
    npm install -g firebase-tools
  • 2. Login ke akun Google Anda:
    firebase login
  • 3. Inisialisasi Firebase project di folder:
    firebase init hosting
  • 4. Pilih project Anda: iot-firebase-6c9d2
  • 5. Tentukan folder public: ketik "dist" atau "." sesuai struktur build Anda
  • 6. Deploy website Anda secara instan:
    firebase deploy

Selesai! Anda akan mendapatkan URL Hosting unik https://iot-firebase-6c9d2.web.app dari Firebase.
